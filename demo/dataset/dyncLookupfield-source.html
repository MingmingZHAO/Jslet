    <script type="text/javascript" src="../../loader/jslet.loader.js"></script>
    <script type="text/javascript">
        //jslet is the system defined module name
        jslet.define("animals", "../js/animals.js");
        jslet.require("jslet,animals", function () {
            initialize();
        });

        function initialize() {
            dsAnimals.datasetListener(doDatasetScroll);
        }

        function doDatasetScroll(dsEvent) {
            if (dsEvent.eventType == jslet.data.DatasetEvent.AFTERSCROLL) {
                dsAnimals.renderOptions("powerfulanimal");
            }
        }
    </script>
